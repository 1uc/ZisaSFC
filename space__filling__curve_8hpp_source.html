15template <int N>
16std::bitset<N> binary_digits(double x) {
17 auto digits = std::bitset<N>{};
18
19 assert(x >= 0);
20 assert(x < 1.0);
21
22 for (int i = N - 1; i >= 0; --i) {
23 if (x < 0.5) {
24 digits[i] = false;
25 x = 2.0 * x;
26 } else {
27 digits[i] = true;
28 x = 2.0 * (x - 0.5);
29 }
30 }
31
32 return digits;
33}

38template <int N>
39std::bitset<2 * N> hilbert_index(const std::bitset<2 * N> &coords) {
40 std::bitset<2 * N> index;
41 std::array<unsigned short, 4> sigma = {0, 3, 1, 2};
42
43 for (int i = N - 1; i >= 0; --i) {
44 std::bitset<2> xy;
45 xy[1] = coords[i * 2 + 1];
46 xy[0] = coords[i * 2];
47
48 auto digits = hilbert_transform(xy, sigma);
49
50 index[i * 2 + 1] = digits[1];
51 index[i * 2] = digits[0];
52 }
53
54 return index;
55}

57template <int N>
58std::bitset<2 * N> hilbert_index(double x, double y) {
59 auto x_digits = binary_digits<N>(x);
60 auto y_digits = binary_digits<N>(y);
61
62 auto coords = std::bitset<2 * N>{};
63 for (int i = 0; i < N; ++i) {
64 coords[2 * i] = x_digits[i];
65 coords[2 * i + 1] = y_digits[i];
66 }
67
68 return hilbert_index<N>(coords);
